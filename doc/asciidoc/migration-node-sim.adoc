|GA | GDS

CALL algo.nodeSimilarity(
  null,
  null,
  {
    graph: 'myGraph',
    writeRelationshipType: 'MY_WRITE_REL_TYPE',
    writeProperty: 'myWriteProperty',
    write: true
  }
)
YIELD
  nodesCompared,
  relationships,
  writeMillis,
  iterations,
  p1,
  writeProperty
----
|
[source, cypher]
----
CALL gds.nodeSimilarity.write(
  'myGraph',
  {
    writeRelationshipType: 'MY_WRITE_REL_TYPE',
    writeProperty: 'myWriteProperty'
  }
)
YIELD
  nodesCompared,
  relationships,
  writeMillis,
  ranIterations,
  similarityDistribution AS sd,
  configuration AS conf
RETURN
  nodesCompared,
  relationships,
  writeMillis,
  ranIterations,
  sd.p1 AS p1,
  conf.writeProperty AS writeProperty
