  null,

  loadMillis,
  iterations,

  writeProperty

----

  'myGraph',
  {
    writeProperty: 'myWriteProperty'

  createMillis,
  ranIterations,
  configuration AS conf
RETURN
  nodesCompared,
  relationships,
  writeMillis,
  ranIterations,
  conf.writeProperty AS writeProperty
